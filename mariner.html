## Load required packages
library(data.table)
library(mariner)

## List loop files for each timepoint
bedpeFiles <-
  list.files(path = "data/loops",
             full.names = TRUE)

## Read in data and format as GInteractions
giList <-
  lapply(bedpeFiles, fread) |>
  lapply(as_ginteractions) |>
  setNames(gsub("[^0-9]+", "", bedpeFiles))

                half = "bottom")
